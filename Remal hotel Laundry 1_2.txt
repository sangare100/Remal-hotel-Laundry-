const langSelect = document.getElementById('lang-select');

let storeEntries = JSON.parse(localStorage.getItem('h_store')) || [];
let hangerCount = 0; let foldingCount = 0;

function checkRoomValidity(tower, roomNum) {
const num = parseInt(roomNum);
if(isNaN(num)) return false;
const floor = Math.floor(num / 100);
if(!validRooms[tower][floor]) return false;
return validRooms[tower][floor].includes(num);
}

window.incrementCounter = function(type) {
if (type === 'hanger') { hangerCount++; document.getElementById('count-hanger').innerText = hangerCount; }
if (type === 'folding') { foldingCount++; document.getElementById('count-folding').innerText = foldingCount; }
};
window.decrementCounter = function(type) {
if (type === 'hanger' && hangerCount > 0) { hangerCount--; document.getElementById('count-hanger').innerText = hangerCount; }
if (type === 'folding' && foldingCount > 0) { foldingCount--; document.getElementById('count-folding').innerText = foldingCount; }
};
function resetCounters() { hangerCount = 0; foldingCount = 0; document.getElementById('count-hanger').innerText = "0"; document.getElementById('count-folding').innerText = "0"; }

function broadcastData() {
localStorage.setItem('h_entries', JSON.stringify(entries));
localStorage.setItem('h_store', JSON.stringify(storeEntries));
fetch(https://kv.val.town/v1/objects/${channelKey}`, {
method: 'POST', headers: { 'Content-Type': 'application/json' },
body: JSON.stringify({ entries, storeEntries })
}).catch(() => {});
updateUI();
}

function pullCloudData() {
fetch(https://kv.val.town/v1/objects/${channelKey}`)
.then(res => res.json())
.then(data => {
if (data) {
entries = data.entries || []; storeEntries = data.storeEntries || [];
updateUI();
}
}).catch(() => {});
}
setInterval(pullCloudData, 3000);
